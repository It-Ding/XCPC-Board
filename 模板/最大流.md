const int INF = 0x3f3f3f3f;
int g[maxn][maxn];
int Maxflow;
int Depth[maxn];
int n, m;

    memset(g, 0, sizeof(g));

## BFS处理深度

```c++
bool BFS()
{
    memset(Depth, -1, sizeof(Depth));
    Depth[1] = 0;
    queue<int> s;
    s.push(1);
    while (!s.empty())
    {
        int now = s.front();
        s.pop();
        for (int i = 1; i <= n; ++i)
        {
            if (g[now][i] > 0 && Depth[i] == -1)
            {
                Depth[i] = Depth[now] + 1;
                s.push(i);
            }
        }
    }
    if (Depth[n] > 0)
        return true;
    return false;
}
```

## DFS寻找增广路径

```c++
int DFS(int x, int mx)
{
    if (x == n)
        return mx;
    int a;
    for (int i = 1; i <= n; ++i)
    {
        if (g[x][i] > 0 && Depth[i] == Depth[x] + 1 && (a = DFS(i, min(g[x][i], mx))))
        {
            g[x][i] -= a;
            g[i][x] += a;
            return a;
        }
    }
    return 0;
}
```
